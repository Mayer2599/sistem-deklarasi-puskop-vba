Private Sub CBCari_Click()
    Dim ws As Worksheet
    Dim searchValue As String
    Dim foundCell As Range
    Dim searchRange As Range
    Dim rowIndex As Long
    Dim anValue As Variant
    
    ' Referensi ke sheet Deklarasi
    Set ws = ThisWorkbook.Sheets("Deklarasi")
    
    ' Ambil nilai dari TextBox LblNAKket
    searchValue = LblNAKket.Text
    
    ' Validasi input
    If Len(searchValue) = 0 Then
        MsgBox "Masukkan nomor yang ingin dicari!", vbExclamation, "Input Kosong"
        Exit Sub
    End If
    
    ' Tentukan range pencarian (kolom E)
    Set searchRange = ws.Range("E7:E400000")
    
    ' Cari nilai di kolom E
    Set foundCell = searchRange.Find(What:=searchValue, LookIn:=xlValues, LookAt:=xlWhole)
    
    ' Jika nilai ditemukan
    If Not foundCell Is Nothing Then
        ' Dapatkan nomor baris dari hasil pencarian
        rowIndex = foundCell.Row
    ' Ambil nilai dari kolom AN
        anValue = ws.Cells(rowIndex, "AN").value
        
        ' Logika untuk menampilkan/menyembunyikan LblADit dan LblADitket
        If Len(LblNAKket.Text) > 0 And Len(anValue) = 0 Then
            ' Jika LblNAKket terisi dan sel AN kosong, sembunyikan LblADit dan LblADitket
            LblADit.Visible = False
            LblADitket.Visible = False
            LblADitket.Enabled = False
        ElseIf Len(LblNAKket.Text) > 0 And Len(anValue) > 0 Then
            ' Jika LblNAKket terisi dan sel AK ada isinya, tampilkan LblADit dan LblADitket
            LblADit.Visible = True
            LblADitket.Visible = True
            LblADitket.Enabled = True
            'Isi nilai dari kolom AN ke LblADitket
            LblADitket.Text = anValue
        End If
        
        ' Isi label dengan data dari baris yang sesuai di sheet Deklarasi
        LblNAkey.Caption = ws.Cells(rowIndex, "G").value ' Kolom G
        LblTLkey.Caption = ws.Cells(rowIndex, "H").value ' Kolom H
        LblJKkey.Caption = ws.Cells(rowIndex, "I").value ' Kolom I
        LblCUPkey.Caption = ws.Cells(rowIndex, "C").value ' Kolom C
        LblNTPkey.Caption = ws.Cells(rowIndex, "D").value ' Kolom D
        
        ' Contoh: Memformat angka dengan pemisah ribuan untuk label LblPin1key
        numericValue = ws.Cells(rowIndex, "J").value ' Ambil nilai dari kolom J
        If IsNumeric(numericValue) Then
            LblPin1key.Caption = Format(numericValue, "#,##0") ' Format dengan pemisah ribuan
            LblPin1key.Caption = Replace(Format(numericValue, "#,##0"), ".", ",")
        Else
            LblPin1key.Caption = "" ' Jika bukan angka, kosongkan caption
        End If
                
        ' Contoh: Memformat angka dengan pemisah ribuan untuk label LblPin1key
        numericValue = ws.Cells(rowIndex, "K").value ' Ambil nilai dari kolom K
        If IsNumeric(numericValue) Then
            LblSim1key.Caption = Format(numericValue, "#,##0") ' Format dengan pemisah ribuan
            LblSim1key.Caption = Replace(Format(numericValue, "#,##0"), ".", ",")
        Else
            LblSim1key.Caption = "" ' Jika bukan angka, kosongkan caption
        End If
        
        ' Contoh: Memformat angka dengan pemisah ribuan untuk label LblPin1key
        numericValue = ws.Cells(rowIndex, "L").value ' Ambil nilai dari kolom L
        If IsNumeric(numericValue) Then
            LblSol1key.Caption = Format(numericValue, "#,##0") ' Format dengan pemisah ribuan
            LblSol1key.Caption = Replace(Format(numericValue, "#,##0"), ".", ",")
        Else
            LblSol1key.Caption = "" ' Jika bukan angka, kosongkan caption
        End If
        
        LblNAPket.Caption = ws.Cells(rowIndex, "AO").value ' Kolom AO
        LblNASket.Caption = ws.Cells(rowIndex, "AP").value ' Kolom AP
        LblNASolket.Caption = ws.Cells(rowIndex, "AQ").value ' Kolom AQ
        
        ' Isi TextBox dengan data dari kolom Q hingga U di sheet Deklarasi
        With ThisWorkbook.Sheets("Deklarasi")
            LblTIket.Text = .Cells(rowIndex, "Q").value ' Kolom Q
            LblTMket.Text = .Cells(rowIndex, "R").value ' Kolom R
            LblTKket.Text = .Cells(rowIndex, "S").value ' Kolom S
            LblTKdstjket.Text = .Cells(rowIndex, "T").value ' Kolom T
            LblTKdbyrket.Text = .Cells(rowIndex, "U").value ' Kolom U
            
            ' Isi CheckBox berdasarkan nilai di kolom V hingga AM di sheet Deklarasi
            CbxPeng1.value = (.Cells(rowIndex, "V").value <> 0)
            CbxAn1.value = (.Cells(rowIndex, "W").value <> 0)
            CbxPen1.value = (.Cells(rowIndex, "X").value <> 0)
            CbxDtlk1.value = (.Cells(rowIndex, "Y").value <> 0)
            CbxDstj1.value = (.Cells(rowIndex, "Z").value <> 0)
            CbxDbyr1.value = (.Cells(rowIndex, "AA").value <> 0)
            CbxPeng2.value = (.Cells(rowIndex, "AB").value <> 0)
            CbxAn2.value = (.Cells(rowIndex, "AC").value <> 0)
            CbxPen2.value = (.Cells(rowIndex, "AD").value <> 0)
            CbxDtlk2.value = (.Cells(rowIndex, "AE").value <> 0)
            CbxDstj2.value = (.Cells(rowIndex, "AF").value <> 0)
            CbxDbyr2.value = (.Cells(rowIndex, "AG").value <> 0)
            CbxPeng3.value = (.Cells(rowIndex, "AH").value <> 0)
            CbxAn3.value = (.Cells(rowIndex, "AI").value <> 0)
            CbxPen3.value = (.Cells(rowIndex, "AJ").value <> 0)
            CbxDtlk3.value = (.Cells(rowIndex, "AK").value <> 0)
            CbxDstj3.value = (.Cells(rowIndex, "AL").value <> 0)
            CbxDbyr3.value = (.Cells(rowIndex, "AM").value <> 0)
        End With
        
               
        ' Aktifkan LblTIket, LblTMket, LblTKket, LblTKdstjket, dan LblTKdbyrket
        LblTIket.Enabled = True
        LblTMket.Enabled = True
        LblTKket.Enabled = True
        LblTKdstjket.Enabled = True
        LblTKdbyrket.Enabled = True
        
        ' Aktifkan semua checkbox
        EnableAllCheckBoxes True
        
        ' Aktifkan tombol CBSimpan
        CBSimpan.Enabled = True
        
        MsgBox "Data ditemukan dan berhasil dimuat!", vbInformation, "Pencarian Berhasil"
    Else
        ' Jika nilai tidak ditemukan
        MsgBox "Nomor tidak ditemukan di database.", vbCritical, "Pencarian Gagal"
        
        ' Nonaktifkan LblTIket, LblTMket, LblTKket, LblTKdstjket, dan LblTKdbyrket
        LblTIket.Enabled = False
        LblTMket.Enabled = False
        LblTKket.Enabled = False
        LblTKdstjket.Enabled = False
        LblTKdbyrket.Enabled = False
        
        ' Kosongkan isi LblTIket dan LblTMket jika ada
        LblTIket.Text = ""
        LblTMket.Text = ""
        LblTKket.Text = ""
        LblTKdstjket.Text = ""
        LblTKdbyrket.Text = ""
        
        ' Kosongkan dan nonaktifkan semua checkbox
        ResetCheckBoxes
        
        ' Nonaktifkan tombol CBSimpan
        CBSimpan.Enabled = False
    End If
End Sub

' Fungsi untuk mengonversi nilai dengan pemisah ribuan menjadi angka biasa
Function ConvertToNumber(ByVal value As String) As Double
    ' Hapus pemisah ribuan (koma) dan konversi ke angka
    If IsNumeric(Replace(value, ",", "")) Then
        ConvertToNumber = CDbl(Replace(value, ",", ""))
    Else
        ConvertToNumber = 0 ' Jika nilai tidak valid, kembalikan 0
    End If
End Function

            If Len(LblTKdbyrket.Text) > 0 Then .Cells(rowIndex, "U").value = LblTKdbyrket.Text
    End With

    MsgBox "Data berhasil disimpan ke sheet Deklarasi!", vbInforation, "Simpan Berhasil"
    ' Kosongkan semua kontrol di UserForm
    ResetUserForm
        
End Sub
