Private Sub LblMNA_Click()
    ' Ubah LblMNA menjadi TextBox untuk memasukkan nomor anggota
    LblMNA.Visible = False ' Sembunyikan Label
    LblNAKket.Visible = True ' Tampilkan TextBox
    LblNAKket.Enabled = True ' Pastikan TextBox aktif
    LblNAKket.SetFocus ' Fokus ke TextBox
End Sub

Private Sub LblNAKket_Exit(ByVal Cancel As MSForms.ReturnBoolean)
    ' Kembalikan caption LblMNA jika TextBox kosong
    If Len(LblNAKket.Text) = 0 Then
        LblNAKket.Visible = False ' Sembunyikan TextBox
        LblMNA.Visible = True ' Tampilkan Label
        LblMNA.Caption = "Masukkan Nomor Anggota Koperasi" ' Kembalikan caption
        
        ' Jangan berikan fokus ke kontrol yang disembunyikan
        ' Biarkan fokus berpindah secara alami
    End If
End Sub

Private Sub CBCari_Click()
    Dim ws As Worksheet
    Dim searchValue As String
    Dim foundCell As Range
    Dim searchRange As Range
    Dim rowIndex As Long
    
    ' Referensi ke sheet Deklarasi
    Set ws = ThisWorkbook.Sheets("Deklarasi")
    
    ' Ambil nilai dari TextBox LblNAKket
    searchValue = LblNAKket.Text
    
    ' Validasi input
    If Len(searchValue) = 0 Then
        MsgBox "Masukkan nomor anggota koperasi!", vbExclamation, "Input Kosong"
        Exit Sub
    End If
    
    ' Reset semua kontrol sebelum memuat data baru
    ResetAllControls
    
    ' Tentukan range pencarian (kolom E)
    Set searchRange = ws.Range("E7:E800000")
    
    ' Cari nilai di kolom E
    Set foundCell = searchRange.Find(What:=searchValue, LookIn:=xlValues, LookAt:=xlWhole)
    
    ' Jika nilai ditemukan
    If Not foundCell Is Nothing Then
        ' Dapatkan nomor baris dari hasil pencarian
        rowIndex = foundCell.Row
        
        ' Muat data dari sheet "Deklarasi" ke UserForm
        With ws
            ' Isi label dengan data dari baris yang sesuai di sheet Deklarasi
            LblNAkey.Caption = .Cells(rowIndex, "G").Value ' Kolom G
            LblTLkey.Caption = .Cells(rowIndex, "H").Value ' Kolom H
            LblJKkey.Caption = .Cells(rowIndex, "I").Value ' Kolom I
            LblCUPkey.Caption = .Cells(rowIndex, "C").Value ' Kolom C
            LblNTPkey.Caption = .Cells(rowIndex, "D").Value ' Kolom D
            
            ' Hitung total pinjaman, simpanan, dan solduka
            LblPin1key.Caption = Application.WorksheetFunction.SumIfs(.Range("M7:M800000"), .Range("J7:J800000"), "Pinjaman", .Range("E7:E800000"), searchValue)
            LblSim1key.Caption = Application.WorksheetFunction.SumIfs(.Range("M7:M800000"), .Range("J7:J800000"), "Simpanan", .Range("E7:E800000"), searchValue)
            LblSol1key.Caption = Application.WorksheetFunction.SumIfs(.Range("M7:M800000"), .Range("J7:J800000"), "Solduka", .Range("E7:E800000"), searchValue)
            
            ' Isi TextBox dengan data dari kolom P, Q, R di sheet Deklarasi
            If .Cells(rowIndex, "P").Value <> "" Then
                LblTIket.Text = .Cells(rowIndex, "P").Value ' Kolom P
            Else
                LblTIket.Text = ""
            End If
            
            If .Cells(rowIndex, "Q").Value <> "" Then
                LblTMket.Text = .Cells(rowIndex, "Q").Value ' Kolom Q
            Else
                LblTMket.Text = ""
            End If
            
            If .Cells(rowIndex, "R").Value <> "" Then
                LblTKket.Text = .Cells(rowIndex, "R").Value ' Kolom R
            Else
                LblTKket.Text = ""
            End If
        End With
        
        ' Reset CheckBox terkait OBMDDR, OBMDDRS, dan OBMDkK
        CBFka.Visible = False
        CBKk.Visible = False
        CBIp.Visible = False
        CBFCkaCUP.Visible = False
        CBFCsspt.Visible = False
        CBFCit.Visible = False
        CBFCiw.Visible = False
        CBSKk.Visible = False
        FZN.Visible = False
        CBSk.Visible = False
        CBSKkB.Visible = False

        ' Aktifkan CheckBox grup berdasarkan caption label
        EnableCheckBoxGroupsBasedOnLabels

        ' Aktifkan semua tools yang dapat diaktifkan
        LblTIket.Enabled = True
        LblTMket.Enabled = True
        LblTKket.Enabled = True

End If
